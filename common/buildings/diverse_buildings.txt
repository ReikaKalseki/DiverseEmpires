building_food_reactor_1 = {
	tags = { food_reactor }

	base_buildtime = @b2time
	
	cost = {
		minerals = @b2cost
	}
	
	produced_resources = {
		energy = 8
	}
	
	required_resources = {
		food = 10
	}
	
	upgrades = {
		building_food_reactor_2
	}	

	prerequisites = {
		#"tech_eco_simulation"
	}
	
	show_tech_unlock_if = { owner = { has_country_flag = dekron } }
	
	potential = {
		owner = { has_country_flag = dekron }
		planet = {
			NOT = { is_planet_class = pc_habitat }	
		}
	}

	destroy_if = {
		exists = owner
		owner = { has_authority = auth_machine_intelligence }
	}
	
	ai_allow = {
		NOR = {
			has_resource = {
				type = sr_betharian
				amount > 0
			}	
			has_resource = {
				type = sr_alien_pets
				amount > 0
			}				
		}
	}

	ai_weight = {
		factor = 1
	}
}
